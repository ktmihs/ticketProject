## 3. 상태 선택 결정 트리

```
상태를 어디에 저장할까?
│
├─ 이 데이터가 여러 컴포넌트에서 필요한가?
│  │
│  ├─ No → 지역 상태 (useState)
│  │        예: 입력값, hover, 모달 열림
│  │
│  └─ Yes
│     │
│     ├─ 이 데이터가 서버에서 왔는가?
│     │  │
│     │  ├─ Yes → React Query
│     │  │         예: 이벤트 목록, 좌석 정보
│     │  │
│     │  └─ No → RTK
│     │           예: 로그인 상태, 대기열 상태
│     │
│     └─ 이 데이터가 실시간으로 변하는가?
│        │
│        ├─ Yes → SSE/WS + RTK
│        │         예: 대기열 순번, 좌석 상태
│        │
│        └─ No → RTK
│                 예: 구매 단계, 선택한 좌석
```

## 4. 실전 예시 코드

### 예시 1: 좌석 선택 페이지

```typescript
// SeatSelectionPage.tsx

import { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useSelector, useDispatch } from 'react-redux';

export default function SeatSelectionPage({ showId }: Props) {
  const dispatch = useDispatch();
  const queryClient = useQueryClient();

  // ═══════════════════════════════════════
  // 1. 지역 상태 (UI만 관련) ✨
  // ═══════════════════════════════════════
  const [hoveredSeat, setHoveredSeat] = useState<number | null>(null);
  const [showPaymentModal, setShowPaymentModal] = useState(false);

  // ═══════════════════════════════════════
  // 2. 전역 상태 (RTK) ✨
  // ═══════════════════════════════════════
  const { selectedSeat, holdToken, holdExpiresAt } = useSelector(
    (state: RootState) => state.purchase
  );

  const { queueToken } = useSelector(
    (state: RootState) => state.queue
  );

  // ═══════════════════════════════════════
  // 3. 서버 상태 (React Query) ✨
  // ═══════════════════════════════════════
  const { data: seats, isLoading } = useQuery({
    queryKey: ['seats', showId],
    queryFn: () => getSeats(showId),
    staleTime: 5000,
    enabled: !!queueToken // 대기열 통과한 사람만
  });

  // ═══════════════════════════════════════
  // 4. Mutation (서버 상태 변경)
  // ═══════════════════════════════════════
  const holdMutation = useMutation({
    mutationFn: (seatId: number) => holdSeat(seatId),

    onMutate: async (seatId) => {
      // Optimistic Update
      await queryClient.cancelQueries({ queryKey: ['seats', showId] });

      const previousSeats = queryClient.getQueryData(['seats', showId]);

      queryClient.setQueryData(['seats', showId], (old: Seat[]) =>
        old.map(seat =>
          seat.id === seatId
            ? { ...seat, status: 'HOLDING' }
            : seat
        )
      );

      return { previousSeats };
    },

    onSuccess: (data) => {
      // RTK 업데이트
      dispatch(purchase.actions.setHold({
        selectedSeat: data.seat,
        holdToken: data.holdToken,
        holdExpiresAt: data.expiresAt
      }));
    },

    onError: (err, variables, context) => {
      // Rollback
      if (context?.previousSeats) {
        queryClient.setQueryData(['seats', showId], context.previousSeats);
      }
      alert('좌석 선택 실패');
    }
  });

  // ═══════════════════════════════════════
  // 5. 실시간 업데이트 (WebSocket)
  // ═══════════════════════════════════════
  useEffect(() => {
    const ws = new WebSocket(`/seats/${showId}/updates`);

    ws.onmessage = (event) => {
      const update = JSON.parse(event.data);

      queryClient.setQueryData(['seats', showId], (old: Seat[]) =>
        old.map(seat =>
          seat.id === update.seatId
            ? { ...seat, status: update.status }
            : seat
        )
      );
    };

    return () => ws.close();
  }, [showId]);

  // ═══════════════════════════════════════
  // 6. 렌더링
  // ═══════════════════════════════════════
  return (
    <div>
      <HoldTimer expiresAt={holdExpiresAt} />

      <SeatMap>
        {seats?.map(seat => (
          <Seat
            key={seat.id}
            seat={seat}
            isHovered={hoveredSeat === seat.id} {/* 지역 */}
            isSelected={selectedSeat?.id === seat.id} {/* 전역 */}
            onMouseEnter={() => setHoveredSeat(seat.id)} {/* 지역 */}
            onMouseLeave={() => setHoveredSeat(null)} {/* 지역 */}
            onClick={() => {
              if (seat.status === 'AVAILABLE') {
                holdMutation.mutate(seat.id);
              }
            }}
          />
        ))}
      </SeatMap>

      {selectedSeat && (
        <button onClick={() => setShowPaymentModal(true)}>
          결제하기
        </button>
      )}

      {/* 지역 상태로 모달 제어 */}
      {showPaymentModal && (
        <PaymentModal
          onClose={() => setShowPaymentModal(false)}
          holdToken={holdToken} {/* 전역 상태 전달 */}
        />
      )}
    </div>
  );
}
```

## 5. 상태 동기화 패턴

### 패턴 A: 서버 → React Query → RTK

```typescript
// 서버에서 데이터를 받아 캐싱한 후, 필요한 부분만 RTK로 복사

const { data } = useQuery({
	queryKey: ['user', 'profile'],
	queryFn: getUserProfile,
});

useEffect(() => {
	if (data) {
		// React Query → RTK 동기화
		dispatch(
			auth.actions.setUser({
				id: data.id,
				name: data.name,
				// 민감 정보는 RTK에 저장하지 않음
			}),
		);
	}
}, [data]);
```

### 패턴 B: RTK → React Query 무효화

```typescript
// RTK 상태 변경 시 관련 React Query 캐시 무효화

const dispatch = useDispatch();
const queryClient = useQueryClient();

const handleLogout = () => {
	dispatch(auth.actions.logout());

	// 모든 사용자 관련 캐시 제거
	queryClient.removeQueries({ queryKey: ['user'] });
	queryClient.removeQueries({ queryKey: ['purchases'] });
};
```

### 패턴 C: SSE → RTK → React Query

```typescript
// 실시간 업데이트를 RTK로 받아 React Query 캐시도 갱신

useEffect(() => {
	const eventSource = new EventSource('/queue/status');

	eventSource.onmessage = event => {
		const data = JSON.parse(event.data);

		// 1. RTK 업데이트 (UI 즉시 반영)
		dispatch(queue.actions.updatePosition(data.position));

		// 2. React Query도 갱신 (일관성)
		queryClient.setQueryData(['queue', 'status'], data);
	};

	return () => eventSource.close();
}, []);
```
