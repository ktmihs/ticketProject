# 상태 전이 다이어그램 (State Machine)

## 1. Queue 상태 머신

### 상태 정의

```typescript
type QueueStatus =
	| 'IDLE' // 초기 상태 (대기열 진입 전)
	| 'JOINING' // 대기열 진입 중
	| 'WAITING' // 대기 중
	| 'ALLOWED' // 구매 허용됨
	| 'EXPIRED' // 시간 만료
	| 'BLOCKED' // 차단됨 (부정 행위)
	| 'ERROR'; // 에러 발생

type QueueErrorReason =
	| 'JOIN_FAILED' // 대기열 진입 API 실패
	| 'SSE_DISCONNECTED' // SSE 연결 끊김
	| 'SERVER_ERROR'; // 서버 5xx 오류

interface QueueState {
	status: QueueStatus;
	position: number | null;
	estimatedWaitTime: number | null;
	queueToken: string | null;
	allowedUntil: number | null;
	error: {
		message: string;
		reason: QueueErrorReason;
	} | null;
}
```

**상태별 설명:**

- **IDLE**: 초기 상태, 대기열 진입 전
- **JOINING**: 대기열 진입 API 호출 중
- **WAITING**: 대기 중, SSE로 순번 업데이트 수신
- **ALLOWED**: 구매 허용됨, 10분간 유효
- **EXPIRED**: 허용 시간 만료, 재진입 필요
- **BLOCKED**: 비정상 요청 또는 정책 위반 시 차단
  - **BLOCKED 상태는 서버 판정 기반이며, 클라이언트 상태 초기화 또는 새로고침으로는 해제되지 않는다.**
  - 해제는 서버 측 차단 해제 또는 관리자 승인 필요
- **ERROR**: 대기열 진입 또는 유지 과정에서 발생한 오류 상태
  - 오류의 상세 원인은 `error.reason`으로 구분한다

**[Queue Error Reason]**

| Reason             | 설명                 | 복구 전략                |
| ------------------ | -------------------- | ------------------------ |
| `JOIN_FAILED`      | 대기열 진입 API 실패 | 재시도 버튼 제공         |
| `SSE_DISCONNECTED` | SSE 연결 끊김        | 자동 재연결 시도         |
| `SERVER_ERROR`     | 서버 5xx 오류        | 일정 시간 후 재시도 안내 |

### RTK Slice 구현

```typescript
// store/slices/queueSlice.ts
import { createSlice, PayloadAction } from '@reduxjs/toolkit';

type QueueStatus =
	| 'IDLE'
	| 'JOINING'
	| 'WAITING'
	| 'ALLOWED'
	| 'EXPIRED'
	| 'BLOCKED'
	| 'ERROR';

type QueueErrorReason = 'JOIN_FAILED' | 'SSE_DISCONNECTED' | 'SERVER_ERROR';

interface QueueState {
	status: QueueStatus;
	showId: string | null;
	position: number | null;
	estimatedWaitTime: number | null;
	queueToken: string | null;
	allowedUntil: number | null;
	error: {
		message: string;
		reason: QueueErrorReason;
	} | null;
}

const initialState: QueueState = {
	status: 'IDLE',
	showId: null,
	position: null,
	estimatedWaitTime: null,
	queueToken: null,
	allowedUntil: null,
	error: null,
};

const queueSlice = createSlice({
	name: 'queue',
	initialState,
	reducers: {
		// IDLE → JOINING
		joinQueue: (state, action: PayloadAction<{ showId: string }>) => {
			state.status = 'JOINING';
			state.showId = action.payload.showId;
			state.error = null;
		},

		// JOINING → WAITING (성공)
		joinSuccess: (
			state,
			action: PayloadAction<{
				position: number;
				estimatedWaitTime: number;
			}>,
		) => {
			state.status = 'WAITING';
			state.position = action.payload.position;
			state.estimatedWaitTime = action.payload.estimatedWaitTime;
		},

		// JOINING → ERROR (실패) - reason 필수
		joinFailure: (
			state,
			action: PayloadAction<{
				message: string;
				reason: QueueErrorReason;
			}>,
		) => {
			state.status = 'ERROR';
			state.error = {
				message: action.payload.message,
				reason: action.payload.reason,
			};
		},

		// WAITING → WAITING (순번 갱신)
		updatePosition: (
			state,
			action: PayloadAction<{
				position: number;
				estimatedWaitTime: number;
			}>,
		) => {
			if (state.status !== 'WAITING') return; // 가드

			state.position = action.payload.position;
			state.estimatedWaitTime = action.payload.estimatedWaitTime;
		},

		// WAITING → ALLOWED
		allowPurchase: (
			state,
			action: PayloadAction<{
				queueToken: string;
				allowedUntil: number;
			}>,
		) => {
			state.status = 'ALLOWED';
			state.queueToken = action.payload.queueToken;
			state.allowedUntil = action.payload.allowedUntil;
			state.position = 0;
		},

		// ALLOWED → EXPIRED (시간 만료)
		expireToken: state => {
			state.status = 'EXPIRED';
			state.queueToken = null;
			state.allowedUntil = null;
		},

		// ANY → BLOCKED (서버 판정)
		blockUser: (state, action: PayloadAction<{ reason: string }>) => {
			state.status = 'BLOCKED';
			state.error = {
				message: action.payload.reason,
				reason: 'SERVER_ERROR', // BLOCKED는 별도 reason
			};
			// BLOCKED 상태는 클라이언트에서 해제 불가
			// 서버 API 호출 시 403으로 계속 차단됨
		},

		// ANY → IDLE (초기화) - BLOCKED 제외
		resetQueue: state => {
			if (state.status === 'BLOCKED') {
				console.warn('BLOCKED 상태는 클라이언트에서 초기화 불가');
				return;
			}
			return initialState;
		},

		// ERROR → IDLE (재시도) - reason별 전략
		retry: state => {
			if (state.status !== 'ERROR') return;

			const canRetry =
				state.error?.reason !== 'SERVER_ERROR' ||
				Date.now() - lastRetryTime > 30000; // 30초 제한

			if (canRetry) {
				state.status = 'IDLE';
				state.error = null;
			}
		},
	},
});

export const {
	joinQueue,
	joinSuccess,
	joinFailure,
	updatePosition,
	allowPurchase,
	expireToken,
	blockUser,
	resetQueue,
	retry,
} = queueSlice.actions;

export default queueSlice.reducer;
```

### SSE 이벤트 핸들러 (Error Reason 처리)

```typescript
// hooks/useQueueSSE.ts
function useQueueSSE() {
	const dispatch = useDispatch();
	const { status, showId } = useSelector((state: RootState) => state.queue);

	useEffect(() => {
		if (status !== 'WAITING' || !showId) return;

		const eventSource = new EventSource(`/api/queue/stream/${showId}`);

		eventSource.addEventListener('position_update', event => {
			const data = JSON.parse(event.data);
			dispatch(
				updatePosition({
					position: data.position,
					estimatedWaitTime: data.estimatedWaitTime,
				}),
			);
		});

		eventSource.addEventListener('allowed', event => {
			const data = JSON.parse(event.data);
			dispatch(
				allowPurchase({
					queueToken: data.queueToken,
					allowedUntil: data.allowedUntil,
				}),
			);
		});

		eventSource.addEventListener('blocked', event => {
			const data = JSON.parse(event.data);
			dispatch(blockUser({ reason: data.reason }));
			eventSource.close(); // BLOCKED는 재연결 불가
		});

		eventSource.onerror = error => {
			console.error('SSE 연결 오류:', error);

			// SSE 끊김 = SSE_DISCONNECTED
			dispatch(
				joinFailure({
					message: '실시간 연결이 끊어졌습니다',
					reason: 'SSE_DISCONNECTED',
				}),
			);

			// 자동 재연결 시도 (3초 후)
			setTimeout(() => {
				if (status === 'ERROR') {
					dispatch(retry());
					dispatch(joinQueue({ showId }));
				}
			}, 3000);
		};

		return () => eventSource.close();
	}, [status, showId, dispatch]);
}
```

### Error Reason별 UI 처리

```typescript
// components/QueueErrorDisplay.tsx
function QueueErrorDisplay() {
  const { error } = useSelector((state: RootState) => state.queue);
  const dispatch = useDispatch();

  if (!error) return null;

  const errorConfig = {
    JOIN_FAILED: {
      title: '대기열 진입 실패',
      message: '일시적인 오류입니다. 다시 시도해주세요.',
      action: '재시도',
      canRetry: true
    },
    SSE_DISCONNECTED: {
      title: '연결 끊김',
      message: '자동으로 재연결을 시도합니다...',
      action: null,
      canRetry: false // 자동 재연결
    },
    SERVER_ERROR: {
      title: '서버 오류',
      message: '잠시 후 다시 시도해주세요. (30초 후 재시도 가능)',
      action: '30초 후 재시도',
      canRetry: false // 시간 제한
    }
  };

  const config = errorConfig[error.reason];

  return (
    <div className="error-box">
      <h3>{config.title}</h3>
      <p>{error.message}</p>
      {config.canRetry && (
        <button onClick={() => dispatch(retry())}>
          {config.action}
        </button>
      )}
    </div>
  );
}
```

## 2. Purchase 상태 머신

### 상태 정의

```typescript
type PurchaseStep =
	| 'IDLE' // 초기 (구매 시작 전)
	| 'QUANTITY_SELECT' // 수량 선택 (좌석 미지정)
	| 'SEAT_SELECT' // 좌석 선택 (좌석 지정)
	| 'SEAT_HOLDING' // 좌석 선점 중
	| 'PAYMENT' // 결제 진행 중
	| 'PROCESSING' // 결제 처리 중
	| 'COMPLETE' // 구매 완료
	| 'FAILED' // 구매 실패
	| 'HOLD_EXPIRED'; // HOLD 만료 (좌석 지정)

interface PurchaseState {
	step: PurchaseStep;
	showId: string | null;
	seatType: 'reserved' | 'non_reserved' | null;

	// 좌석 미지정
	quantity: number;

	// 좌석 지정
	selectedSeat: {
		id: number;
		row: string;
		number: number;
		price: number;
	} | null;
	holdToken: string | null;
	holdExpiresAt: number | null; // 서버 기준 만료 시각

	// 공통
	totalAmount: number;
	paymentMethod: string | null;
	purchaseId: string | null;
	error: string | null;
}
```

**[Purchase State Machine 진입 조건]**

Purchase 상태 머신은 **Queue.status === 'ALLOWED'** 인 경우에만 진입할 수 있다.

- Queue ALLOWED 상태 검증은 미들웨어에서 수행
- ALLOWED가 아닌 상태에서 구매 시도 시 대기열 페이지로 리다이렉트

### RTK Slice 구현

```typescript
// store/slices/purchaseSlice.ts
import { createSlice, PayloadAction } from '@reduxjs/toolkit';

type PurchaseStep =
	| 'IDLE'
	| 'QUANTITY_SELECT'
	| 'SEAT_SELECT'
	| 'SEAT_HOLDING'
	| 'PAYMENT'
	| 'PROCESSING'
	| 'COMPLETE'
	| 'FAILED'
	| 'HOLD_EXPIRED';

interface PurchaseState {
	step: PurchaseStep;
	showId: string | null;
	seatType: 'reserved' | 'non_reserved' | null;

	// 좌석 미지정
	quantity: number;

	// 좌석 지정
	selectedSeat: {
		id: number;
		row: string;
		number: number;
		price: number;
	} | null;
	holdToken: string | null;
	holdExpiresAt: number | null;

	// 공통
	totalAmount: number;
	paymentMethod: string | null;
	purchaseId: string | null;
	error: string | null;
}

const initialState: PurchaseState = {
	step: 'IDLE',
	showId: null,
	seatType: null,
	quantity: 1,
	selectedSeat: null,
	holdToken: null,
	holdExpiresAt: null,
	totalAmount: 0,
	paymentMethod: null,
	purchaseId: null,
	error: null,
};

const purchaseSlice = createSlice({
	name: 'purchase',
	initialState,
	reducers: {
		// IDLE → QUANTITY_SELECT or SEAT_SELECT
		// ⚠️ 이 액션은 Queue ALLOWED 상태에서만 호출되어야 함
		startPurchase: (
			state,
			action: PayloadAction<{
				showId: string;
				seatType: 'reserved' | 'non_reserved';
			}>,
		) => {
			state.showId = action.payload.showId;
			state.seatType = action.payload.seatType;

			if (action.payload.seatType === 'non_reserved') {
				state.step = 'QUANTITY_SELECT';
			} else {
				state.step = 'SEAT_SELECT';
			}
		},

		// QUANTITY_SELECT → PAYMENT (좌석 미지정)
		selectQuantity: (
			state,
			action: PayloadAction<{
				quantity: number;
				pricePerTicket: number;
			}>,
		) => {
			state.quantity = action.payload.quantity;
			state.totalAmount =
				action.payload.quantity * action.payload.pricePerTicket;
			state.step = 'PAYMENT';
		},

		// SEAT_SELECT → SEAT_HOLDING (좌석 지정)
		requestSeatHold: (
			state,
			action: PayloadAction<{
				seatId: number;
			}>,
		) => {
			state.step = 'SEAT_HOLDING';
		},

		// SEAT_HOLDING → PAYMENT (성공)
		holdSeatSuccess: (
			state,
			action: PayloadAction<{
				seat: {
					id: number;
					row: string;
					number: number;
					price: number;
				};
				holdToken: string;
				expiresAt: number; // 서버가 반환한 만료 시각
			}>,
		) => {
			state.step = 'PAYMENT';
			state.selectedSeat = action.payload.seat;
			state.holdToken = action.payload.holdToken;
			state.holdExpiresAt = action.payload.expiresAt; // 서버 기준 저장
			state.totalAmount = action.payload.seat.price;
		},

		// SEAT_HOLDING → SEAT_SELECT (실패)
		holdSeatFailure: (state, action: PayloadAction<{ error: string }>) => {
			state.step = 'SEAT_SELECT';
			state.error = action.payload.error;
		},

		// PAYMENT → PROCESSING
		startPayment: (
			state,
			action: PayloadAction<{
				paymentMethod: string;
			}>,
		) => {
			state.step = 'PROCESSING';
			state.paymentMethod = action.payload.paymentMethod;
			state.error = null;
		},

		// PROCESSING → COMPLETE
		completePayment: (
			state,
			action: PayloadAction<{
				purchaseId: string;
			}>,
		) => {
			state.step = 'COMPLETE';
			state.purchaseId = action.payload.purchaseId;
		},

		// PROCESSING → FAILED
		failPayment: (state, action: PayloadAction<{ error: string }>) => {
			state.step = 'FAILED';
			state.error = action.payload.error;
		},

		// FAILED → PAYMENT (재시도)
		retryPayment: state => {
			state.step = 'PAYMENT';
			state.error = null;
		},

		// PAYMENT/PROCESSING → HOLD_EXPIRED (좌석 지정만)
		// ⚠️ 이 액션은 서버 응답 또는 클라이언트 타이머에서 발생
		expireHold: state => {
			if (state.seatType !== 'reserved') return;

			state.step = 'HOLD_EXPIRED';
			state.holdToken = null;
			state.holdExpiresAt = null;
			state.selectedSeat = null;
		},

		// HOLD_EXPIRED → SEAT_SELECT
		returnToSeatSelect: state => {
			state.step = 'SEAT_SELECT';
			state.error = null;
		},

		// ANY → IDLE (초기화)
		resetPurchase: () => initialState,
	},
});

export const {
	startPurchase,
	selectQuantity,
	requestSeatHold,
	holdSeatSuccess,
	holdSeatFailure,
	startPayment,
	completePayment,
	failPayment,
	retryPayment,
	expireHold,
	returnToSeatSelect,
	resetPurchase,
} = purchaseSlice.actions;

export default purchaseSlice.reducer;
```

```typescript
// middleware/queueGuard.ts
import { Middleware } from '@reduxjs/toolkit';
import { startPurchase } from '../slices/purchaseSlice';

export const queueGuard: Middleware = store => next => action => {
	const state = store.getState();

	// Purchase 시작 시 Queue ALLOWED 상태 확인
	if (action.type === startPurchase.type) {
		const queueStatus = state.queue.status;

		if (queueStatus !== 'ALLOWED') {
			console.error(
				`Purchase 진입 차단: Queue 상태가 ALLOWED가 아님 (현재: ${queueStatus})`,
			);

			// 대기열 페이지로 리다이렉트
			window.location.href = '/queue';
			return; // 액션 차단
		}

		// Queue 토큰 만료 확인
		const allowedUntil = state.queue.allowedUntil;
		if (allowedUntil && Date.now() >= allowedUntil) {
			console.error('Queue 토큰 만료됨');
			window.location.href = '/queue';
			return;
		}
	}

	return next(action);
};
```

```typescript
// hooks/useHoldExpiry.ts
function useHoldExpiry() {
	const dispatch = useDispatch();
	const { step, holdExpiresAt, seatType } = useSelector(
		(state: RootState) => state.purchase,
	);

	useEffect(() => {
		if (seatType !== 'reserved') return;
		if (!['PAYMENT', 'PROCESSING'].includes(step)) return;
		if (!holdExpiresAt) return;

		// 클라이언트 타이머 (UI 표시용)
		const checkExpiry = setInterval(() => {
			const timeLeft = holdExpiresAt - Date.now();

			if (timeLeft <= 0) {
				dispatch(expireHold());
				alert('좌석 선점 시간이 만료되었습니다. 다시 선택해주세요.');
				clearInterval(checkExpiry);
			}
		}, 1000);

		return () => clearInterval(checkExpiry);
	}, [step, holdExpiresAt, seatType, dispatch]);

	// 남은 시간 반환 (UI 표시용)
	const [timeLeft, setTimeLeft] = useState<number | null>(null);

	useEffect(() => {
		if (!holdExpiresAt) return;

		const updateTimer = setInterval(() => {
			setTimeLeft(Math.max(0, holdExpiresAt - Date.now()));
		}, 100);

		return () => clearInterval(updateTimer);
	}, [holdExpiresAt]);

	return { timeLeft };
}
```

```typescript
async function purchaseReserved(req, res) {
	const { holdToken } = req.body;

	// 서버에서 holdToken 검증 + 만료 확인
	const decoded = jwt.verify(holdToken, SECRET);

	// 서버 기준 만료 체크

	});

	if (!seat) {
		return res.status(410).json({
			error: 'HOLD_EXPIRED',
			message: '좌석 선점이 해제되었습니다',
		});
	}
```

```typescript
describe('Queue State Machine', () => {
	it('IDLE → JOINING → WAITING', () => {
		const initialState = { status: 'IDLE', position: null };

		let state = queueReducer(initialState, joinQueue({ showId: 'show123' }));
		expect(state.status).toBe('JOINING');

		state = queueReducer(
			state,
			joinSuccess({ position: 100, estimatedWaitTime: 300 }),
		);
		expect(state.status).toBe('WAITING');
		expect(state.position).toBe(100);
	});

	it('JOINING → ERROR with reason', () => {
		const state = { status: 'JOINING' as const };

		const newState = queueReducer(
			state,
			joinFailure({
				message: 'API 서버 오류',
				reason: 'SERVER_ERROR',
			}),
		);
```

### ✅ Queue 상태 전이 규칙

| From    | Event          | To      | 조건          | 비고                     |
| ------- | -------------- | ------- | ------------- | ------------------------ |
| IDLE    | joinQueue      | JOINING | -             | -                        |
| JOINING | joinSuccess    | WAITING | API 성공      | -                        |
| JOINING | joinFailure    | ERROR   | API 실패      | reason 필수              |
| WAITING | updatePosition | WAITING | SSE 이벤트    | -                        |
| WAITING | allowPurchase  | ALLOWED | position = 0  | -                        |
| ALLOWED | expireToken    | EXPIRED | 10분 경과     | -                        |
| ALLOWED | resetQueue     | IDLE    | 구매 완료     | -                        |
| ERROR   | retry          | IDLE    | reason별 전략 | SERVER_ERROR는 30초 제한 |
| ANY     | blockUser      | BLOCKED | 서버 판정     | **클라이언트 해제 불가** |
